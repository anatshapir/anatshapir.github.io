public static void what2(BinNode<Integer> bt, Node<Integer> lst) {
    if (bt != null) {
        int x = bt.getValue();
        if (what3(bt.getLeft(), x) && what3(bt.getRight(), x))
            lst = new Node<Integer>(x, lst);
        what2(bt.getLeft(), lst);
        what2(bt.getRight(), lst);
    }
}

public static boolean what3(BinNode<Integer> bt, int x) {
    if (bt == null)         return true;
    if (bt.getValue() == x) return false;
    return what3(bt.getLeft(), x) && what3(bt.getRight(), x);
}

public static boolean secret(BinNode<Integer> bt) {
    if (bt.getLeft() == null && bt.getRight() == null)
        return true;
    if (bt.getLeft() == null || bt.getRight() == null)
        return false;
    if (bt.getLeft().getValue() > 0)  return false;
    if (bt.getRight().getValue() < 0) return false;
    return secret(bt.getLeft()) && secret(bt.getRight());
}

public static boolean result(BinNode<Integer> bt, int a, int b) {
    if (bt == null) return false;
    if (bt.getValue() == a)
        return sod(bt.getLeft(), b) || sod(bt.getRight(), b);
    return result(bt.getLeft(), a, b) || result(bt.getRight(), a, b);
}

public static boolean sod(BinNode<Integer> bt, int k) {
    if (bt == null) return false;
    int x = bt.getValue();
    if (x == k)   return true;
    return sod(bt.getLeft(), k) || sod(bt.getRight(), k);
}

    int x2 = what(bt.getRight());

public static boolean what(BinNode<Integer> bt) {
    if (bt.getLeft() == null && bt.getRight() == null)
        return true;
    if (bt.getLeft() == null) {
        if (bt.getValue() >= bt.getRight().getValue())
            return what(bt.getRight());
        return false;
    }
    if (bt.getRight() == null)
        return bt.getValue() >= bt.getLeft().getValue()
               && what(bt.getLeft());
    return false;
}

        return 0;

        return 2 + what(bt.getLeft()) + what(bt.getRight());

public static boolean what(BinNode<Integer> bt) {
    if (bt == null) return true;
    int x = bt.getValue();
    if ((bt.getLeft()  != null && bt.getLeft().getValue()  >= x) ||
        (bt.getRight() != null && bt.getRight().getValue() >= x))
        return false;
    return what(bt.getLeft()) && what(bt.getRight());
}

public static void doWhat(BinNode<Integer> bt,
                           Stack<Integer> stk, int data) {
    if (bt != null) {
        int x = bt.getValue();
        if (x == data) {
            System.out.print(x + " ");
            while (!stk.isEmpty()) {
                x = stk.pop();
                System.out.print(x + " ");
            }
            System.out.println();
        } else {
            stk.push(x);
            doWhat(bt.getLeft(), stk, data);
            if (!stk.isEmpty()) doWhat(bt.getRight(), stk, data);
            if (!stk.isEmpty()) stk.pop();
        }
    }
}

public static void doWhat(BinNode<Integer> bt) {
    BinNode<Integer> temp = bt;
    Stack<BinNode<Integer>> s1 = new Stack<>();
    Stack<Integer> s2 = new Stack<>();
    int m=0; double a=0; int lev=0; int lvl=lev;
    s1.push(temp); s2.push(lev);
    while (!s1.isEmpty()) {
        temp = s1.pop(); lev = s2.pop();
        if (lvl < lev) lvl = lev;
        m++;
        a = (a*(m-1) + temp.getValue()) / m;
        if (temp.getLeft()  != null) { s1.push(temp.getLeft());  s2.push(lev+1); }
        if (temp.getRight() != null) { s1.push(temp.getRight()); s2.push(lev+1); }
    }
    System.out.println(a+" "+m+" "+lvl+" "+lev);
}

public static void mystery(BinNode<Integer> bt) {
    BinNode<Integer> node = bt;
    Stack<BinNode<Integer>> stk = new Stack<>();
    do {
        while (node != null) {
            stk.push(node);
            node = node.getLeft();
        }
        if (!stk.isEmpty()) {
            node = stk.pop();
            System.out.print(node.getValue() + " ");
            node = node.getRight();
        }
    } while (!stk.isEmpty() || node != null);
}

public static boolean isSpecial(BinNode<Integer> bt, int x) {
    if (bt == null)         return true;
    if (bt.getValue() > x) return false;
    return isSpecial(bt.getLeft(),  x+1)
        && isSpecial(bt.getRight(), x+1);
}
